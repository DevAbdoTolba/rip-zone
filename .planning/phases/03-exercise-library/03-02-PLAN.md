    ```typescript
    import { Suspense } from 'react'

    import musclesData from '@/../data/muscles.json'
    import warmupsData from '@/../data/warmups.json'

    export default function ExercisesPage() {
      return (
        <Suspense fallback={<div className="p-4 text-muted-foreground text-[16px]">Loading exercises...</div>}>

            muscles={musclesData}
            warmups={warmupsData}
          />
        </Suspense>
      )
    }
    ```

    - `'use client'` directive at top
    - Props: `exercises` (array from exercises.json), `muscles` (array from muscles.json), `warmups` (array from warmups.json)

    - Muscle group display order: `['chest', 'back', 'shoulders', 'arms', 'forearms', 'core', 'legs', 'glutes', 'calves']` per D-01
    - Muscle group display names: `{ chest: 'Chest', back: 'Back', shoulders: 'Shoulders', arms: 'Arms', forearms: 'Forearms', core: 'Core', legs: 'Legs', glutes: 'Glutes', calves: 'Calves' }`
    - When search/filter is active (any of query, equipment, difficulty, muscleFilter is non-empty), sections auto-expand and empty sections hide completely
    - Render structure:
      ```
      <div className="flex flex-col min-h-screen">
        <div className="sticky top-0 z-20 bg-background">
          <div className="px-8 pt-6 pb-2">
            <h1 className="text-[28px] font-semibold text-foreground mb-4">Exercises</h1>
          </div>
          <ExerciseFilters ... />
        </div>
        <div className="px-8 py-4">
          {/* Muscle group sections */}
          {MUSCLE_GROUP_ORDER.map(group => {
            const groupExercises = groupedExercises.get(group)
            if (hasActiveFilter && (!groupExercises || groupExercises.length === 0)) return null
            const isExpanded = expandedSections.has(group)
            return (
              <section key={group} className="mb-12">
                <button onClick={toggleSection(group)} className="flex items-center justify-between w-full py-2">
                  <h2 className="text-[20px] font-semibold text-foreground">{GROUP_LABELS[group]}</h2>
                  <ChevronDown className={cn('text-muted-foreground transition-transform', !isExpanded && 'rotate-180')} size={20} />
                </button>
                {isExpanded && (
                  <div className="flex flex-col gap-3 mt-3">
                    {groupExercises?.map(ex => (
                      <ExerciseCard
                        key={ex.slug}
                        exercise={ex}
                        muscles={muscles}
                        warmups={warmups}
                        isLevel1={expandedCards.has(ex.slug)}
                        isLevel2={level2Card === ex.slug}
                        onToggleLevel1={() => toggleLevel1(ex.slug)}
                        onToggleLevel2={() => toggleLevel2(ex.slug)}
                        onOpenWarmup={() => openWarmup(ex)}
                      />
                    ))}
                  </div>
                )}
              </section>
            )
          })}
          {/* Empty state per D-15 */}
          {filteredExercises.length === 0 && (
            <div className="flex flex-col items-center justify-center py-16 text-center">
              <img
                src="https://media.giphy.com/media/3oEjI6SIIHBdRxXI40/giphy.gif"

                Those filters are stricter than your gym bro's diet. Try clearing them — we've got 110 exercises waiting.

            </div>
          )}
        </div>
      </div>
      ```

        // Page title visible

        await expect(page.getByRole('heading', { name: 'Legs' })).toBeVisible()
      })

      test('EXER-01: exercise card expands to Level 1 on click', async ({ page }) => {
        // Click first exercise card
        const firstCard = page.locator('[class*="bg-card"]').first()
        await firstCard.click()
        // Level 1 content should be visible (description, badges)

        await expect(firstCard.locator('text=More')).toBeVisible()
      })

      test('EXER-01: exercise card expands to Level 2 with form cues', async ({ page }) => {
        // Click first card to Level 1
        const firstCard = page.locator('[class*="bg-card"]').first()
        await firstCard.click()
        // Click "More" to Level 2
        await firstCard.getByText('More').click()
        // Level 2 content should appear
        await expect(firstCard.locator('text=Form Cues').or(firstCard.locator('text=Secondary Muscles'))).toBeVisible()
        // Less button should be visible
        await expect(firstCard.getByText('Less')).toBeVisible()
      })

      test('EXER-02: search filters exercises by name', async ({ page }) => {
        // Type into search bar
        const searchInput = page.getByPlaceholder('Search exercises')
        await searchInput.fill('bench')
        // Wait for filtering
        await page.waitForTimeout(100)
        // Visible exercise cards should contain "bench" in their text
        const cards = page.locator('[class*="bg-card"]')
        const count = await cards.count()
        expect(count).toBeGreaterThan(0)
        expect(count).toBeLessThan(110) // filtered down from 110
        // First visible card should contain "bench" (case-insensitive)
        const firstName = await cards.first().textContent()
        expect(firstName?.toLowerCase()).toContain('bench')
      })

      test('EXER-02: equipment filter chips work', async ({ page }) => {
        // Click "Barbell" filter chip
        await page.getByText('Barbell', { exact: true }).click()

        // Should have filtered results (fewer than 110)
        // Click first card to expand and check equipment badge
        const firstCard = page.locator('[class*="bg-card"]').first()
        await firstCard.click()
        await expect(firstCard.getByText('Barbell')).toBeVisible()

        await page.getByText('Clear filters').click()
        await page.waitForTimeout(100)
        // Exercise cards should be visible again
        await expect(page.locator('[class*="bg-card"]').first()).toBeVisible()
      })

        await firstCard.click()

        await expect(page.getByText('1').first()).toBeVisible()
        // Close button should work

        // Sheet should be dismissed
        await expect(page.getByText(/Warm-up$/).first()).not.toBeVisible()
      })

      test('bottom nav has Map and Exercises tabs', async ({ page }) => {

      })
    })

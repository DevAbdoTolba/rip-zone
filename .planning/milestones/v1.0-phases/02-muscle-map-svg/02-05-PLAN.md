    **2. Create `src/components/muscle-map/DisambiguationZoom.tsx`:**

    ```typescript
    'use client'

    import { useMapStore } from '@/stores/useMapStore'
    import { MUSCLE_CENTROIDS, MUSCLE_DISPLAY_NAMES } from './muscle-clusters'
    import type { MuscleSlug } from '@/types'

    export function DisambiguationZoom() {
      const { zoomRegion, selectMuscle, setZoomRegion } = useMapStore()

      if (!zoomRegion) return null

      const handleMuscleSelect = (slug: MuscleSlug) => {
        selectMuscle(slug)
        setZoomRegion(null) // Auto-zoom back per D-16
      }

      const handleBackdropClick = () => {
        setZoomRegion(null) // Dismiss zoom without selecting
      }

      return (
        <>
          {/* Cluster title label */}
          <text
            x="50"
            y={parseFloat(zoomRegion.viewBox.split(' ')[1]) + 2}
            fill="oklch(0.95 0.01 270)"
            fontSize="4"
            fontWeight="600"
            textAnchor="middle"
          >
            {zoomRegion.label}
          </text>

          {/* Muscle name labels at centroids */}
          {zoomRegion.muscles.map((slug) => {
            const centroid = MUSCLE_CENTROIDS[slug]
            const displayName = MUSCLE_DISPLAY_NAMES[slug] || slug
            if (!centroid) return null
            return (
              <text
                key={slug}
                x={centroid.x}
                y={centroid.y - 3}
                fill="oklch(0.95 0.01 270)"
                fontSize="3"
                fontWeight="400"
                textAnchor="middle"
                style={{ pointerEvents: 'none' }}
              >
                {displayName}
              </text>
            )
          })}

          {/* Backdrop rect for dismissing zoom — covers full SVG area below everything */}
          <rect
            x="0"
            y="0"
            width="100"
            height="250"
            fill="transparent"
            onClick={handleBackdropClick}
            style={{ cursor: 'pointer' }}
          />
        </>
      )
    }
    ```

    NOTE: DisambiguationZoom renders SVG elements (text, rect) — it must be rendered INSIDE the SVG element, not as HTML siblings. The MuscleMapCanvas component will render it inside the SVG via a wrapper approach.

      (e: React.MouseEvent<HTMLDivElement>) => {
        const target = e.target as SVGElement
        const hitElement = target.closest('[id^="hit-"]') as SVGElement | null
        const id = hitElement?.id || (target.id?.startsWith('hit-') ? target.id : null)

        if (!id) return

        // D-15: Disambiguation only in Advanced and Anatomy modes
        if (detailMode !== 'normal' && CLUSTER_MAP[id]) {
          setZoomRegion(CLUSTER_MAP[id])
          return
        }

        // Direct selection — extract slug from hit target ID
        if (id.startsWith('hit-muscle-')) {
          const slug = id.replace('hit-muscle-', '') as MuscleSlug
          if (selectedMuscle === slug) {
            selectMuscle(null)
          } else {
            selectMuscle(slug)
          }
        }
      },
      [selectedMuscle, selectMuscle, detailMode, setZoomRegion]
    )
